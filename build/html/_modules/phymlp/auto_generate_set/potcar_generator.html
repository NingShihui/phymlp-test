def read_poscar_elements(poscar_path):
    """
    正确读取POSCAR文件中的元素种类和顺序
    """
    try:
        # 使用绝对路径确保文件可访问
        abs_poscar_path = os.path.abspath(poscar_path)
        print(f"  读取POSCAR文件: {abs_poscar_path}")

        with open(abs_poscar_path, 'r', encoding='utf-8') as f:
            lines = [line.strip() for line in f.readlines() if line.strip()]

        # 跳过注释行
        line_index = 0
        while line_index < len(lines) and (lines[line_index].startswith('#') or not lines[line_index]):
            line_index += 1

        # 第1行: 系统名称
        system_name = lines[line_index]
        line_index += 1

        # 第2行: 缩放因子
        scaling_factor = lines[line_index]
        line_index += 1

        # 第3-5行: 晶格向量
        lattice_vectors = lines[line_index:line_index + 3]
        line_index += 3

        # 第6行: 元素种类
        elements_line = lines[line_index]
        elements = elements_line.split()
        line_index += 1

        # 第7行: 各元素原子数
        atom_counts_line = lines[line_index]
        atom_counts = list(map(int, atom_counts_line.split()))
        line_index += 1

        print(f"  从POSCAR读取到的元素顺序: {elements}")
        print(f"  各元素原子数: {atom_counts}")

        return elements

    except Exception as e:
        print(f"  读取POSCAR文件时出错: {e}")
        return None

def find_potcar_files(potcar_base_dir, functional, elements, potcar_variants=None):
    """
    查找每个元素对应的POTCAR文件路径
    """
    if potcar_variants is None:
        potcar_variants = {}

    potcar_paths = []
    functional_dir = os.path.join(potcar_base_dir, functional)

    print(f"  在目录中查找赝势文件: {functional_dir}")

    if not os.path.exists(functional_dir):
        print(f"  错误: 泛函目录不存在: {functional_dir}")
        return None

    # 首先检查目录结构
    available_dirs = [d for d in os.listdir(functional_dir) if os.path.isdir(os.path.join(functional_dir, d))]
    print(f"  可用的赝势目录: {available_dirs}")

    for i, element in enumerate(elements):
        # 确定要查找的赝势变体优先级
        variants_to_try = []

        # 1. 首先尝试用户指定的变体
        if element in potcar_variants:
            variants_to_try.append(potcar_variants[element])

        # 2. 然后尝试标准变体
        variants_to_try.extend([element, f"{element}_pv", f"{element}_sv", f"{element}_d"])

        # 去除重复
        variants_to_try = list(dict.fromkeys(variants_to_try))

        found = False
        for variant in variants_to_try:
            potcar_path = os.path.join(functional_dir, variant, "POTCAR")
            if os.path.exists(potcar_path):
                potcar_paths.append(potcar_path)
                print(f"    ✓ {element} -> {variant}/POTCAR")
                found = True
                break
            else:
                print(f"      尝试 {variant}: 文件不存在")

        if not found:
            print(f"  错误: 未找到元素 {element} 的POTCAR文件")
            print(f"    尝试的变体: {variants_to_try}")
            return None

    return potcar_paths

def concatenate_potcar_files(potcar_paths, output_path):
    """
    拼接多个POTCAR文件
    """
    try:
        # 删除已存在的输出文件
        if os.path.exists(output_path):
            os.remove(output_path)

        # 使用Python文件操作拼接
        with open(output_path, 'wb') as outfile:
            for i, potcar_path in enumerate(potcar_paths):
                with open(potcar_path, 'rb') as infile:
                    content = infile.read()
                    outfile.write(content)
                element_name = os.path.basename(os.path.dirname(potcar_path))
                print(f"    ✓ 已添加: {element_name}/POTCAR")

        print(f"  成功生成POTCAR文件: {output_path}")

        # 显示文件信息
        file_size = os.path.getsize(output_path)
        print(f"  文件大小: {file_size} 字节")

        return True

    except Exception as e:
        print(f"  拼接POTCAR文件时出错: {e}")
        return False

def generate_potcar_auto(poscar_path, config):
    """
    自动生成POTCAR文件 - 修复版本
    """
    potcar_config = config['vasp']['potcar']

    potcar_base_dir = potcar_config.get('potcar_base_dir')
    functional = potcar_config.get('functional', 'PBE')
    potcar_variants = potcar_config.get('potcar_variants', {})
    output_path = "POTCAR"

    print("  使用自动POTCAR生成器...")
    print(f"  赝势根目录: {potcar_base_dir}")
    print(f"  泛函类型: {functional}")

    # 检查POTCAR目录是否存在
    if not os.path.exists(potcar_base_dir):
        print(f"  错误: 赝势根目录不存在: {potcar_base_dir}")
        return False

    # 使用绝对路径读取POSCAR
    abs_poscar_path = os.path.abspath(poscar_path)
    if not os.path.exists(abs_poscar_path):
        print(f"  错误: POSCAR文件不存在: {abs_poscar_path}")
        return False

    # 读取POSCAR中的元素顺序
    elements = read_poscar_elements(abs_poscar_path)
    if not elements:
        return False

    # 查找POTCAR文件
    potcar_paths = find_potcar_files(potcar_base_dir, functional, elements, potcar_variants)
    if not potcar_paths:
        return False

    # 拼接POTCAR文件
    print(f"  正在拼接POTCAR文件...")
    success = concatenate_potcar_files(potcar_paths, output_path)

    return success

def generate_potcar_pymatgen(structure, config):
    """
    使用pymatgen生成POTCAR
    """
    try:
        from pymatgen.io.vasp import Potcar

        # 获取唯一的元素符号
        elements = []
        for site in structure:
            element = site.specie.symbol
            if element not in elements:
                elements.append(element)

        print(f"  使用pymatgen生成POTCAR元素: {elements}")

        # 使用pymatgen生成POTCAR
        potcar = Potcar(elements)
        return potcar

    except Exception as e:
        print(f"  pymatgen生成POTCAR失败: {e}")
        return None

def generate_potcar(poscar_path, structure, config, output_dir):
    """
    主POTCAR生成函数 - 修复版本
    """
    potcar_config = config['vasp']['potcar']
    method = potcar_config.get('method', 'auto')

    output_path = os.path.join(output_dir, "POTCAR")

    print(f"  生成POTCAR文件...")

    # 保存当前工作目录
    original_cwd = os.getcwd()

    try:
        # 切换到目标目录
        os.chdir(output_dir)
        print(f"  切换到目录: {os.getcwd()}")

        # 检查POSCAR文件在目标目录中是否存在（应该是被复制过来的）
        local_poscar = "POSCAR"
        if not os.path.exists(local_poscar):
            print(f"  错误: 目标目录中不存在POSCAR文件")
            print(f"  当前目录内容: {os.listdir('.')}")
            return False

        if method == 'auto':
            # 使用自动POTCAR生成器，但使用本地POSCAR文件
            success = generate_potcar_auto(local_poscar, config)
            return success

        elif method == 'pymatgen':
            # 使用pymatgen
            potcar = generate_potcar_pymatgen(structure, config)
            if potcar:
                potcar.write_file("POTCAR")
                return True
            else:
                return False

        elif method == 'manual':
            # 手动模式 - 需要用户提前准备好POTCAR
            potcar_dir = potcar_config.get('potcar_dir')
            if potcar_dir and os.path.exists(potcar_dir):
                print(f"  手动模式: 请确保POTCAR文件已准备好")
                return True
            else:
                print(f"  错误: 手动模式需要设置有效的potcar_dir")
                return False

        else:
            print(f"  错误: 未知的POTCAR生成方法: {method}")
            return False

    except Exception as e:
        print(f"  生成POTCAR时出错: {e}")
        import traceback
        traceback.print_exc()
        return False
    finally:
        # 恢复原始工作目录
        os.chdir(original_cwd)
