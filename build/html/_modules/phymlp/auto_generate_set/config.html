"""
配置管理模块

该模块负责加载和管理整个项目的配置信息，包括Materials Project API配置、
路径设置、形变参数、微扰参数、VASP计算参数等。
"""

import yaml
import os
import glob

def load_config(config_file="input.yaml"):
    """加载和验证配置文件。

    Args:
        config_file (str): 配置文件的路径，默认为 'input.yaml'。

    Returns:
        dict: 包含所有配置参数的字典。

    Raises:
        FileNotFoundError: 当指定的配置文件不存在时抛出。
        yaml.YAMLError: 当配置文件格式错误时抛出。

    Examples:
        >>> config = load_config("input.yaml")
        >>> api_key = config['api']['key']
        >>> elements = config['elements']
    """
    if not os.path.exists(config_file):
        raise FileNotFoundError(f"配置文件 '{config_file}' 不存在!")

    with open(config_file, 'r', encoding='utf-8') as f:
        config = yaml.safe_load(f)

    # 设置默认值
    config = _set_default_values(config)

    return config




def _set_default_values(config):
    """为配置参数设置默认值。

    Args:
        config (dict): 原始配置字典。

    Returns:
        dict: 包含默认值的完整配置字典。
    """
    # 设置路径默认值
    config.setdefault('paths', {})
    config['paths'].setdefault('base_dir', './materials_project')
    config['paths'].setdefault('cif_dir', 'base/dir_CIF')
    config['paths'].setdefault('poscar_dir', 'base/dir_POSCAR')
    config['paths'].setdefault('deform_dir', 'deform')
    config['paths'].setdefault('vasp_dir', 'vasp_calculations')
    config['paths'].setdefault('birch_murnaghan_dir', 'Birch_Murnaghan_fit')
    config['paths'].setdefault('extxyz_set_dir', 'Extxyz_set')

    # 设置用户自定义POSCAR默认配置
    config.setdefault('user_poscar', {})
    config['user_poscar'].setdefault('enabled', False)
    config['user_poscar'].setdefault('source_dir', '')

    # 设置形变参数默认值
    config.setdefault('deformation', {})
    config['deformation'].setdefault('max_atoms', 150)
    config['deformation'].setdefault('scaling_factors', [0.8, 0.9, 1.0, 1.1, 1.2])
    config['deformation'].setdefault('max_supercell', 4)

    # 设置微扰参数默认值
    config.setdefault('perturbation', {})
    config['perturbation'].setdefault('max_atoms', 150)
    config['perturbation'].setdefault('max_supercell', 4)
    config['perturbation'].setdefault('n_structures', 100)
    config['perturbation'].setdefault('cell_pert_fraction', 0.05)
    config['perturbation'].setdefault('atom_pert_distance', 0.05)
    config['perturbation'].setdefault('atom_pert_style', 'uniform')

    # 设置VASP计算默认参数
    config.setdefault('vasp', {})
    config['vasp'].setdefault('incar_parameters', {})
    config['vasp'].setdefault('potcar', {})
    config['vasp'].setdefault('submission', {})

    # 设置INCAR参数默认值
    default_incar = {
        'ALGO': 'normal',
        'EDIFF': 1E-3,
        'KSPACING': 0.2,
        'ENCUT': 400,
        'ISIF': 2,
        'ISMEAR': 1,
        'NELM': 400,
        'NSW': 0,
        'PREC': 'Accurate',
        'SIGMA': 0.1,
        'NCORE': 32,
        'KPAR': 2,
        'LREAL': 'Auto'
    }

    for key, value in default_incar.items():
        config['vasp']['incar_parameters'].setdefault(key, value)

    # 设置POTCAR默认配置
    config['vasp']['potcar'].setdefault('method', 'auto')
    config['vasp']['potcar'].setdefault('functional', 'PBE')
    config['vasp']['potcar'].setdefault('potcar_variants', {})
    config['vasp']['potcar'].setdefault('potcar_base_dir', './POTCAR/')

    # 设置提交脚本默认配置
    default_submission = {
        'slurm_headers': [
            '#SBATCH -N 1',
            '#SBATCH -n 32',
            '#SBATCH -J NSH-test',
            '#SBATCH -p xahcnormal'
        ],
        'modules': [
            'module purge',
            'module load vasp-6.3.2-intelmpi2017_optcell'
        ],
        'environment': [
            'export MKL_DEBUG_CPU_TYPE=5',
            'export MKL_CBWR=AVX2',
            'export I_MPI_PIN_DOMAIN=numa'
        ],
        'vasp_execution': [
            'srun --mpi=pmi2 vasp_std'
        ],
        'max_batch_size': 10,
        'check_interval': 60
    }

    for key, value in default_submission.items():
        config['vasp']['submission'].setdefault(key, value)

    # 设置Birch-Murnaghan拟合默认配置
    config.setdefault('birch_murnaghan_fitting', {})
    bm_fitting = config['birch_murnaghan_fitting']
    bm_fitting.setdefault('source', {})
    bm_fitting['source'].setdefault('root_dir', './materials_project/vasp_calculations/deform/')
    bm_fitting['source'].setdefault('search_recursive', True)

    bm_fitting.setdefault('results', {})
    bm_fitting['results'].setdefault('root_dir', './Birch_Murnaghan_fit')
    bm_fitting['results'].setdefault('search_recursive', False)

    bm_fitting.setdefault('data_filter', {})
    bm_fitting['data_filter'].setdefault('pressure_range', [-10000, 10000])
    bm_fitting['data_filter'].setdefault('energy_range', [-10000, 1000])

    bm_fitting.setdefault('fitting', {})
    bm_fitting['fitting'].setdefault('initial_guess', {})
    bm_fitting['fitting']['initial_guess'].setdefault('E0', 'auto')
    bm_fitting['fitting']['initial_guess'].setdefault('V0', 'auto')
    bm_fitting['fitting']['initial_guess'].setdefault('B0', 100)
    bm_fitting['fitting']['initial_guess'].setdefault('B1', 4.0)

    bm_fitting['fitting'].setdefault('optimization', {})
    bm_fitting['fitting']['optimization'].setdefault('max_iterations', 1000)
    bm_fitting['fitting']['optimization'].setdefault('tolerance', 1e-6)

    bm_fitting.setdefault('output', {})
    bm_fitting['output'].setdefault('save_plot', True)
    bm_fitting['output'].setdefault('plot_format', 'png')
    bm_fitting['output'].setdefault('plot_dpi', 300)
    bm_fitting['output'].setdefault('save_parameters', True)
    bm_fitting['output'].setdefault('save_fitted_data', True)

    # 设置Extxyz数据集生成默认配置
    config.setdefault('extxyz_generation', {})
    extxyz_gen = config['extxyz_generation']

    extxyz_gen.setdefault('vasp', {})
    extxyz_gen['vasp'].setdefault('outcar_search_dir', './materials_project/vasp_calculations')
    extxyz_gen['vasp'].setdefault('search_recursive', True)
    extxyz_gen['vasp'].setdefault('extraction_step', 1)

    extxyz_gen.setdefault('birch_murnaghan', {})
    extxyz_gen['birch_murnaghan'].setdefault('fitting_results_dir', './Birch_Murnaghan_fit')
    extxyz_gen['birch_murnaghan'].setdefault('search_recursive', False)
    extxyz_gen['birch_murnaghan'].setdefault('volume_range', 'auto')
    extxyz_gen['birch_murnaghan'].setdefault('n_structures_per_fit', 50)
    extxyz_gen['birch_murnaghan'].setdefault('include_original_points', True)
    extxyz_gen['birch_murnaghan'].setdefault('use_outcar_structures', True)
    extxyz_gen['birch_murnaghan'].setdefault('scaling_factors', [0.8, 1.2])

    extxyz_gen['birch_murnaghan'].setdefault('pressure_filter', {})
    extxyz_gen['birch_murnaghan']['pressure_filter'].setdefault('enabled', True)
    extxyz_gen['birch_murnaghan']['pressure_filter'].setdefault('pressure_range', [-1000, 1000])

    return config
