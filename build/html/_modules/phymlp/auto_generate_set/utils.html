def mkdir_output_load(path):
    """创建输出目录，如果已存在则清空"""
    if os.path.exists(path):
        shutil.rmtree(path)
    os.makedirs(path, exist_ok=True)

def chemsys_comb(elements):
    """生成所有可能的元素组合"""
    all_combinations = []
    for r in range(1, len(elements) + 1):
        for combo in combinations(elements, r):
            all_combinations.append('-'.join(combo))
    return all_combinations

def get_formula_from_filename(filename):
    """从文件名中提取化学式"""
    parts = filename.split('-')
    formula_parts = []
    for part in parts:
        if part.startswith('mp'):
            break
        formula_parts.append(part)
    return '-'.join(formula_parts)

def safe_copy(src, dst):
    """安全复制文件"""
    try:
        shutil.copy2(src, dst)
        return True
    except Exception as e:
        print(f"复制文件失败: {src} -> {dst}, 错误: {e}")
        return False

def check_potcar_availability(config):
    """检查POTCAR配置是否可用"""
    potcar_config = config['vasp']['potcar']
    method = potcar_config.get('method', 'auto')

    if method == 'auto':
        potcar_base_dir = potcar_config.get('potcar_base_dir')
        if not potcar_base_dir or not os.path.exists(potcar_base_dir):
            print(f"警告: POTCAR基础目录不存在: {potcar_base_dir}")
            return False
        return True

    elif method == 'pymatgen':
        try:
            from pymatgen.io.vasp import Potcar
            # 简单测试pymatgen是否能工作
            test_potcar = Potcar(['H'])
            return True
        except Exception as e:
            print(f"警告: pymatgen POTCAR生成不可用: {e}")
            return False

    elif method == 'manual':
        potcar_dir = potcar_config.get('potcar_dir')
        if not potcar_dir or not os.path.exists(potcar_dir):
            print(f"警告: 手动模式POTCAR目录不存在: {potcar_dir}")
            return False
        return True

    else:
        print(f"警告: 未知的POTCAR生成方法: {method}")
        return False

def safe_path_join(*args):
    """安全的路径连接，处理混合的路径分隔符"""
    # 将所有参数转换为字符串，并替换混合的分隔符
    clean_args = []
    for arg in args:
        if arg is None:
            continue
        # 将混合的分隔符统一为系统分隔符
        clean_arg = str(arg).replace('\\', '/').replace('//', '/')
        clean_args.append(clean_arg)

    # 使用os.path.join连接路径
    joined_path = os.path.join(*clean_args)

    # 再次清理路径
    cleaned_path = os.path.normpath(joined_path)
    return cleaned_path

def check_file_exists(file_path):
    """检查文件是否存在，提供详细的错误信息"""
    if not os.path.exists(file_path):
        print(f"错误: 文件不存在: {file_path}")
        print(f"  当前工作目录: {os.getcwd()}")
        print(f"  文件绝对路径: {os.path.abspath(file_path)}")

        # 检查父目录是否存在
        parent_dir = os.path.dirname(file_path)
        if not os.path.exists(parent_dir):
            print(f"  父目录不存在: {parent_dir}")
        else:
            print(f"  父目录存在，但文件不存在")
            # 列出父目录中的文件
            try:
                files_in_dir = os.listdir(parent_dir)
                print(f"  父目录中的文件: {files_in_dir}")
            except Exception as e:
                print(f"  无法列出父目录内容: {e}")

        return False
    return True
