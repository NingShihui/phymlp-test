def birch_murnaghan_eqn(params, V):
    """
    Birch-Murnaghan物态方程
    
    Args:
        params: [E0, V0, B0, B1] 拟合参数
        V: 体积数组
        
    Returns:
        E: 能量数组
    """
    E0, V0, B0, B1 = params
    eta = (V0 / V) ** (2/3)
    return E0 + (9 * V0 * B0 / 16) * ((eta - 1)**3 * B1 + (eta - 1)**2 * (6 - 4 * eta))

def birch_murnaghan_residuals(params, V, E):
    """
    计算拟合残差
    
    Args:
        params: 拟合参数
        V: 体积数据
        E: 能量数据
        
    Returns:
        residuals: 残差数组
    """
    return birch_murnaghan_eqn(params, V) - E

def read_a_v_e_p_file(file_path):
    """
    读取a_V_E_P.txt文件，处理多空格分隔问题
    
    Args:
        file_path: 文件路径
        
    Returns:
        df: 处理后的DataFrame
    """
    try:
        # 首先尝试读取文件内容
        with open(file_path, 'r') as f:
            lines = f.readlines()
        
        # 处理表头
        header_line = lines[0].strip()
        # 使用多个空格分割表头
        headers = [h.strip() for h in header_line.split() if h.strip()]
        
        data_lines = []
        for line in lines[1:]:
            line = line.strip()
            if line:
                # 使用多个空格分割数据行，并过滤空字符串
                values = [v.strip() for v in line.split() if v.strip()]
                if len(values) == len(headers):
                    data_lines.append(values)
        
        # 创建DataFrame
        df = pd.DataFrame(data_lines, columns=headers)
        
        # 确保所有列都是数值类型
        for col in df.columns:
            df[col] = pd.to_numeric(df[col], errors='coerce')
        
        # 删除包含NaN的行
        original_len = len(df)
        df = df.dropna()
        if len(df) < original_len:
            print(f"  Removed {original_len - len(df)} rows with NaN/invalid values")
        
        print(f"  Successfully read {len(df)} data points from {file_path}")
        return df
        
    except Exception as e:
        print(f"  Error reading file {file_path}: {e}")
        return pd.DataFrame()

def ensure_numeric_parameters(config):
    """
    确保配置参数都是数值类型
    
    Args:
        config: 配置字典
        
    Returns:
        config: 修复后的配置字典
    """
    # 修复拟合容差参数
    if 'fitting' in config and 'optimization' in config['fitting']:
        optim_config = config['fitting']['optimization']
        if 'tolerance' in optim_config:
            optim_config['tolerance'] = float(optim_config['tolerance'])
        if 'max_iterations' in optim_config:
            optim_config['max_iterations'] = int(optim_config['max_iterations'])
    
    # 修复初始猜测参数
    if 'fitting' in config and 'initial_guess' in config['fitting']:
        init_config = config['fitting']['initial_guess']
        if 'B0' in init_config:
            init_config['B0'] = float(init_config['B0'])
        if 'B1' in init_config:
            init_config['B1'] = float(init_config['B1'])
    
    # 修复数据筛选参数
    if 'data_filter' in config:
        filter_config = config['data_filter']
        if 'pressure_range' in filter_config:
            filter_config['pressure_range'] = [float(x) for x in filter_config['pressure_range']]
        if 'energy_range' in filter_config:
            filter_config['energy_range'] = [float(x) for x in filter_config['energy_range']]
        if 'max_energy_deviation' in filter_config:
            filter_config['max_energy_deviation'] = float(filter_config['max_energy_deviation'])
    
    return config

def filter_data(df, filter_config):
    """
    根据配置筛选数据
    
    Args:
        df: 包含V, E, P数据的DataFrame
        filter_config: 筛选配置
        
    Returns:
        filtered_df: 筛选后的DataFrame
    """
    if len(df) == 0:
        return df
    
    filtered_df = df.copy()
    
    # 压力筛选
    if 'pressure_range' in filter_config:
        p_min, p_max = filter_config['pressure_range']
        mask = (filtered_df['P'] >= p_min) & (filtered_df['P'] <= p_max)
        filtered_df = filtered_df[mask]
        print(f"  Pressure filter: {p_min} <= P <= {p_max}, kept {len(filtered_df)} points")
    
    # 能量筛选
    if 'energy_range' in filter_config:
        e_min, e_max = filter_config['energy_range']
        mask = (filtered_df['E'] >= e_min) & (filtered_df['E'] <= e_max)
        filtered_df = filtered_df[mask]
        print(f"  Energy filter: {e_min} <= E <= {e_max}, kept {len(filtered_df)} points")
    
    # 能量偏差筛选
    if 'max_energy_deviation' in filter_config and len(filtered_df) > 0:
        max_dev = filter_config['max_energy_deviation']
        e_mean = filtered_df['E'].mean()
        mask = abs(filtered_df['E'] - e_mean) <= max_dev
        filtered_df = filtered_df[mask]
        print(f"  Energy deviation filter: |E - {e_mean:.4f}| <= {max_dev}, kept {len(filtered_df)} points")
    
    print(f"  Data filtering: {len(df)} -> {len(filtered_df)} points")
    return filtered_df

def get_initial_guess(df, config):
    """
    获取初始猜测参数
    
    Args:
        df: 数据DataFrame
        config: 拟合配置
        
    Returns:
        p0: 初始参数数组 [E0, V0, B0, B1]
    """
    initial_config = config.get('initial_guess', {})
    
    # E0: 自动选择最小值或使用配置值
    if initial_config.get('E0') == 'auto':
        E0 = float(df['E'].min())
    else:
        E0 = float(initial_config.get('E0', df['E'].min()))
    
    # V0: 自动选择最小值对应的体积或使用配置值
    if initial_config.get('V0') == 'auto':
        min_energy_idx = df['E'].idxmin()
        V0 = float(df.loc[min_energy_idx, 'V'])
    else:
        V0 = float(initial_config.get('V0', df['V'].mean()))
    
    # B0和B1使用配置值
    B0 = float(initial_config.get('B0', 100))
    B1 = float(initial_config.get('B1', 4.0))
    
    return [E0, V0, B0, B1]

def perform_fitting(df, config):
    """
    执行Birch-Murnaghan拟合
    
    Args:
        df: 数据DataFrame
        config: 拟合配置
        
    Returns:
        result: 拟合结果字典
    """
    if df.empty:
        print("  No valid data for fitting")
        return None
    
    # 明确转换为numpy数组并指定数据类型
    V = np.array(df['V'].values, dtype=np.float64)
    E = np.array(df['E'].values, dtype=np.float64)
    
    print(f"  Data types - V: {V.dtype}, E: {E.dtype}")
    print(f"  Data ranges - V: [{V.min():.2f}, {V.max():.2f}], E: [{E.min():.2f}, {E.max():.2f}]")
    
    # 获取初始猜测
    p0 = get_initial_guess(df, config)
    print(f"  Initial guess: E0={p0[0]:.4f}, V0={p0[1]:.4f}, B0={p0[2]:.2f}, B1={p0[3]:.2f}")
    
    # 拟合配置
    fit_config = config.get('fitting', {})
    optim_config = fit_config.get('optimization', {})
    
    # 确保容差参数是数值类型
    ftol = float(optim_config.get('tolerance', 1e-6))
    max_nfev = int(optim_config.get('max_iterations', 1000))
    
    print(f"  Fitting parameters - ftol: {ftol}, max_nfev: {max_nfev}")
    
    try:
        # 执行最小二乘拟合
        result = least_squares(
            birch_murnaghan_residuals,
            p0,
            args=(V, E),
            max_nfev=max_nfev,
            ftol=ftol,
            method='trf'  # 使用信任域反射算法，更稳定
        )
        print(f"  Fitting completed: {result.success}")
        if hasattr(result, 'message'):
            print(f"  Fitting message: {result.message}")
        return result
    except Exception as e:
        print(f"  Fitting error: {e}")
        import traceback
        traceback.print_exc()
        return None

def find_scale_directories(source_dir):
    """
    在源目录中查找所有scale_*目录
    
    Args:
        source_dir: 源目录路径
        
    Returns:
        list: 包含(scale目录路径, 缩放系数)的元组列表
    """
    scale_dirs = []
    
    # 查找所有scale_*目录
    scale_patterns = [
        "scale_*",
        "scale*",
        "*scale*"
    ]
    
    for pattern in scale_patterns:
        found_dirs = glob.glob(os.path.join(source_dir, pattern))
        for dir_path in found_dirs:
            if os.path.isdir(dir_path):
                # 从目录名提取缩放系数
                dir_name = os.path.basename(dir_path)
                # 尝试提取数字部分
                import re
                numbers = re.findall(r"[-+]?\d*\.\d+|\d+", dir_name)
                if numbers:
                    scale_factor = float(numbers[0])
                    scale_dirs.append((dir_path, scale_factor))
    
    # 按缩放系数排序
    scale_dirs.sort(key=lambda x: x[1])
    return scale_dirs

def create_outcar_scale_directory(bm_structure_dir, original_data, source_path):
    """
    在Birch-Murnaghan结构目录中创建OUTCAR_scale目录并收集原始OUTCAR文件
    
    Args:
        bm_structure_dir: Birch-Murnaghan结构目录
        original_data: 原始数据DataFrame
        source_path: 原始a_V_E_P.txt文件路径
    """
    outcar_scale_dir = os.path.join(bm_structure_dir, 'OUTCAR_scale')
    os.makedirs(outcar_scale_dir, exist_ok=True)
    
    print(f"  Creating OUTCAR_scale directory: {outcar_scale_dir}")
    
    # 获取源目录（a_V_E_P.txt文件所在的目录）
    source_dir = os.path.dirname(source_path)
    
    found_structures = 0
    
    # 查找所有scale目录
    scale_dirs = find_scale_directories(source_dir)
    
    if scale_dirs:
        print(f"  Found {len(scale_dirs)} scale directories in {source_dir}")
        
        for scale_dir, scale_factor in scale_dirs:
            outcar_file = os.path.join(scale_dir, 'OUTCAR')
            if os.path.exists(outcar_file):
                dest_file = os.path.join(outcar_scale_dir, f'OUTCAR_scale_{scale_factor:.1f}')
                shutil.copy2(outcar_file, dest_file)
                print(f"    Copied: {os.path.basename(scale_dir)}/OUTCAR -> OUTCAR_scale_{scale_factor:.1f}")
                found_structures += 1
            else:
                print(f"    Warning: OUTCAR not found in {scale_dir}")
    else:
        print(f"  No scale directories found in {source_dir}")
        
        # 尝试在父目录中查找
        parent_dir = os.path.dirname(source_dir)
        parent_scale_dirs = find_scale_directories(parent_dir)
        
        if parent_scale_dirs:
            print(f"  Found {len(parent_scale_dirs)} scale directories in parent directory {parent_dir}")
            
            for scale_dir, scale_factor in parent_scale_dirs:
                outcar_file = os.path.join(scale_dir, 'OUTCAR')
                if os.path.exists(outcar_file):
                    dest_file = os.path.join(outcar_scale_dir, f'OUTCAR_scale_{scale_factor:.1f}')
                    shutil.copy2(outcar_file, dest_file)
                    print(f"    Copied: {os.path.basename(scale_dir)}/OUTCAR -> OUTCAR_scale_{scale_factor:.1f}")
                    found_structures += 1
    
    if found_structures == 0:
        print(f"  Warning: Could not find any OUTCAR files")
        
        # 如果没有找到OUTCAR文件，至少保存数据信息
        data_info_file = os.path.join(outcar_scale_dir, 'data_info.txt')
        with open(data_info_file, 'w') as f:
            f.write("Original VASP calculation data information\n")
            f.write("==========================================\n\n")
            f.write(f"Structure: {os.path.basename(bm_structure_dir)}\n")
            f.write(f"Source directory: {source_dir}\n")
            f.write(f"Number of data points: {len(original_data)}\n")
            f.write(f"Volume range: {original_data['V'].min():.2f} - {original_data['V'].max():.2f} Å³\n")
            f.write(f"Energy range: {original_data['E'].min():.2f} - {original_data['E'].max():.2f} eV\n")
            if 'a' in original_data.columns:
                f.write(f"Lattice constant range: {original_data['a'].min():.2f} - {original_data['a'].max():.2f} Å\n")
    
    print(f"  Total OUTCAR files collected: {found_structures}")
    return outcar_scale_dir

def save_fitting_results(result, df, output_dir, config, structure_name, source_path, reference_volume=None):
    """
    保存拟合结果
    
    Args:
        result: 拟合结果
        df: 原始数据
        output_dir: 输出目录
        config: 配置
        structure_name: 结构名称
        source_path: 原始数据文件路径
        reference_volume: 参考体积（用于计算缩放因子）
    """
    if result is None:
        return None
    
    # 拟合参数
    E0, V0, B0, B1 = result.x
    B0_GPa = B0 * 160.21766208  # 转换为GPa
    
    # 计算拟合数据 - 生成更多点（默认1000个）
    n_fit_points = config.get('output', {}).get('n_fit_points', 1000)
    V_min = float(df['V'].min())
    V_max = float(df['V'].max())
    V_fit = np.linspace(V_min * 0.95, V_max * 1.05, n_fit_points)
    E_fit = birch_murnaghan_eqn(result.x, V_fit)
    
    # 计算缩放因子（如果提供了参考体积）
    scale_factors = None
    if reference_volume is not None:
        scale_factors = (V_fit / V0) ** (1/3)
    
    # 计算残差和R²
    E_pred = birch_murnaghan_eqn(result.x, df['V'].values)
    residuals = E_pred - df['E'].values
    ss_res = np.sum(residuals**2)
    ss_tot = np.sum((df['E'].values - np.mean(df['E'].values))**2)
    r_squared = 1 - (ss_res / ss_tot) if ss_tot != 0 else 0
    
    # 保存拟合参数
    if config.get('output', {}).get('save_parameters', True):
        param_file = os.path.join(output_dir, 'fitting_parameters.txt')
        with open(param_file, 'w') as f:
            f.write("Birch-Murnaghan Fitting Parameters\n")
            f.write("==================================\n\n")
            f.write(f"Structure: {structure_name}\n\n")
            f.write(f"Equilibrium Energy (E0): {E0:.6f} eV\n")
            f.write(f"Equilibrium Volume (V0): {V0:.6f} Å³\n")
            f.write(f"Bulk Modulus (B0): {B0_GPa:.4f} GPa\n")
            f.write(f"Pressure Derivative (B1): {B1:.4f}\n\n")
            f.write(f"R-squared: {r_squared:.6f}\n")
            f.write(f"Root Mean Square Error: {np.sqrt(ss_res/len(df)):.6f} eV\n")
            f.write(f"Number of data points: {len(df)}\n")
            f.write(f"Number of fitting points: {n_fit_points}\n")
            f.write(f"Optimization success: {result.success}\n")
            f.write(f"Number of iterations: {result.nfev}\n")
            if hasattr(result, 'message'):
                f.write(f"Optimization message: {result.message}\n")
    
    # 保存拟合数据
    if config.get('output', {}).get('save_fitted_data', True):
        fit_data_file = os.path.join(output_dir, 'fitted_data.csv')
        if scale_factors is not None:
            fit_df = pd.DataFrame({
                'scale_factor': scale_factors,
                'Volume_fit': V_fit,
                'Energy_fit': E_fit
            })
        else:
            fit_df = pd.DataFrame({
                'Volume_fit': V_fit,
                'Energy_fit': E_fit
            })
        fit_df.to_csv(fit_data_file, index=False)
        print(f"  Saved {len(fit_df)} fitting points to fitted_data.csv")
    
    # 创建OUTCAR_scale目录
    if config.get('output', {}).get('create_outcar_scale_dir', True):
        outcar_scale_dir = create_outcar_scale_directory(output_dir, df, source_path)
        outcar_files = glob.glob(os.path.join(outcar_scale_dir, 'OUTCAR_scale_*'))
        print(f"  Created OUTCAR_scale directory with {len(outcar_files)} OUTCAR files")
    
    # 绘制拟合曲线
    if config.get('output', {}).get('save_plot', True):
        plt.figure(figsize=(10, 8))
        
        # 散点图：体积 vs 能量
        plt.scatter(df['V'], df['E'], color='red', s=60, alpha=0.8, 
                   label='VASP computed data', zorder=5)
        
        # 拟合曲线
        plt.plot(V_fit, E_fit, 'b-', linewidth=2.5, 
                label='Birch-Murnaghan Fitting', zorder=4)
        
        # 标记平衡点
        plt.plot(V0, E0, 'go', markersize=10, markeredgecolor='black', 
                label=f'Equilibrium (V0={V0:.2f} Å³, E0={E0:.2f} eV)', zorder=6)
        
        plt.xlabel('Volume (Å³)', fontsize=14, fontweight='bold')
        plt.ylabel('Energy (eV)', fontsize=14, fontweight='bold')
        plt.title(f'Birch-Murnaghan Equation of State Fitting\n{structure_name}', 
                 fontsize=16, fontweight='bold', pad=20)
        plt.legend(fontsize=12, framealpha=0.9)

        # 添加拟合参数文本
        textstr = '\n'.join([
            f'E0 = {E0:.4f} eV',

            f'B1 = {B1:.3f}',
            f'R² = {r_squared:.4f}',
            f'Fit points = {n_fit_points}'
        ])
        props = dict(boxstyle='round', facecolor='lightblue', alpha=0.8)

                      verticalalignment='top', bbox=props, fontfamily='monospace')
        
        plot_format = config.get('output', {}).get('plot_format', 'png')

def find_a_v_e_p_files(root_dir, recursive=True):
    """
    在目录树中查找所有的a_V_E_P.txt文件
    
    Args:
        root_dir: 根目录
        recursive: 是否递归搜索
        
    Returns:
        list: 包含(a_V_E_P.txt文件路径, 相对路径)的元组列表
    """
    a_v_e_p_files = []
    
    if recursive:
        # 递归搜索所有子目录
        for root, dirs, files in os.walk(root_dir):
            for file in files:
                if file == 'a_V_E_P.txt':
                    file_path = os.path.join(root, file)
                    # 计算相对于root_dir的相对路径
                    rel_path = os.path.relpath(root, root_dir)
                    a_v_e_p_files.append((file_path, rel_path))
    else:
        # 只在根目录搜索
        for item in os.listdir(root_dir):
            item_path = os.path.join(root_dir, item)
            if os.path.isdir(item_path):
                data_file = os.path.join(item_path, 'a_V_E_P.txt')
                if os.path.exists(data_file):
                    a_v_e_p_files.append((data_file, item))
    
    return a_v_e_p_files

def create_mirror_structure(source_files, bm_dir):
    """
    根据找到的a_V_E_P.txt文件创建镜像目录结构
    
    Args:
        source_files: 找到的a_V_E_P.txt文件列表
        bm_dir: Birch-Murnaghan拟合目录
        
    Returns:
        dict: 创建的目录结构信息
    """
    structure_info = {}
    
    # 创建主目录
    os.makedirs(bm_dir, exist_ok=True)
    
    for file_path, rel_path in source_files:
        # 如果相对路径是当前目录('.')，则使用文件所在目录名
        if rel_path == '.':
            dir_name = os.path.basename(os.path.dirname(file_path))
        else:
            # 使用相对路径作为目录名，替换路径分隔符为下划线
            dir_name = rel_path.replace(os.path.sep, '_')
        
        # 创建对应的BM目录
        bm_item_dir = os.path.join(bm_dir, dir_name)
        os.makedirs(bm_item_dir, exist_ok=True)
        
        # 复制a_V_E_P.txt文件
        dest_file = os.path.join(bm_item_dir, 'a_V_E_P.txt')
        shutil.copy2(file_path, dest_file)
        
        structure_info[dir_name] = {
            'source_path': file_path,
            'bm_path': bm_item_dir,
            'relative_path': rel_path
        }
        
        print(f"Created mirror directory: {dir_name}")
        print(f"  Source: {file_path}")
        print(f"  Target: {dest_file}")
    
    return structure_info

def get_reference_volume(source_path):
    """
    从源目录获取参考体积
    
    Args:
        source_path: 原始数据文件路径
        
    Returns:
        float: 参考体积
    """
    try:
        # 尝试从POSCAR文件获取参考结构
        source_dir = os.path.dirname(source_path)
        poscar_file = os.path.join(source_dir, 'POSCAR')
        
        if os.path.exists(poscar_file):
            from ase.io import read
            atoms = read(poscar_file)
            return atoms.get_volume()
        
        # 如果没有POSCAR，尝试从scale_1.0目录获取
        scale_1_dir = os.path.join(source_dir, 'scale_1.0')
        if os.path.exists(scale_1_dir):
            poscar_file = os.path.join(scale_1_dir, 'POSCAR')
            if os.path.exists(poscar_file):
                from ase.io import read
                atoms = read(poscar_file)
                return atoms.get_volume()
        
        # 如果都没有，使用原始数据的平均体积
        df = read_a_v_e_p_file(source_path)
        if len(df) > 0:
            return df['V'].mean()
        
    except Exception as e:
        print(f"  Warning: Could not get reference volume: {e}")
    
    return None

def process_all_birch_murnaghan_fitting(config_path):
    """
    处理所有Birch-Murnaghan拟合
    
    Args:
        config_path: 配置文件路径
    """
    # 加载配置
    with open(config_path, 'r') as f:
        config = yaml.safe_load(f)
    
    paths = config['paths']
    bm_config = config.get('birch_murnaghan_fitting', {})
    
    # 确保所有配置参数都是数值类型
    bm_config = ensure_numeric_parameters(bm_config)
    
    source_config = bm_config.get('source', {})
    source_root_dir = source_config.get('root_dir', './deform')
    search_recursive = source_config.get('search_recursive', True)
    
    # 使用配置中的拟合结果目录，如果没有则使用paths中的默认目录
    results_config = bm_config.get('results', {})
    bm_dir = results_config.get('root_dir', paths.get('birch_murnaghan_dir', 'Birch_Murnaghan_fit'))
    
    if not os.path.exists(source_root_dir):
        print(f"Error: Source directory '{source_root_dir}' not found!")
        return
    
    print("Step 1: Searching for a_V_E_P.txt files...")
    a_v_e_p_files = find_a_v_e_p_files(source_root_dir, search_recursive)
    
    if not a_v_e_p_files:
        print(f"No a_V_E_P.txt files found in '{source_root_dir}'")
        return
    
    print(f"Found {len(a_v_e_p_files)} a_V_E_P.txt files")
    
    print("Step 2: Creating mirror directory structure...")
    structure_info = create_mirror_structure(a_v_e_p_files, bm_dir)
    
    print("Step 3: Performing Birch-Murnaghan fitting...")
    
    successful_fits = 0
    total_fits = len(structure_info)
    
    # 遍历所有BM目录进行拟合
    for structure_name, info in structure_info.items():
        bm_item_dir = info['bm_path']
        data_file = os.path.join(bm_item_dir, 'a_V_E_P.txt')
        
        print(f"\nProcessing: {structure_name}")
        print(f"  Source: {info['source_path']}")
        
        try:
            # 读取数据
            df = read_a_v_e_p_file(data_file)
            
            if len(df) == 0:
                print(f"  ✗ No valid data found in {data_file}")
                continue
            
            print(f"  Loaded {len(df)} data points")
            print(f"  Data columns: {list(df.columns)}")
            
            # 数据筛选
            filtered_df = filter_data(df, bm_config.get('data_filter', {}))
            
            if len(filtered_df) < 4:
                print(f"  ✗ Not enough data points after filtering ({len(filtered_df)} < 4)")
                continue
            
            # 获取参考体积
            reference_volume = get_reference_volume(info['source_path'])
            if reference_volume:
                print(f"  Reference volume: {reference_volume:.2f} Å³")
            
            # 执行拟合
            result = perform_fitting(filtered_df, bm_config)
            
            if result is None:
                print(f"  ✗ Fitting failed")
                continue
            
            # 保存结果
            fitting_results = save_fitting_results(result, filtered_df, bm_item_dir, bm_config, 
                                                 structure_name, info['source_path'], reference_volume)
            
            if fitting_results and fitting_results['success']:
                params = fitting_results['parameters']
                stats = fitting_results['statistics']
                print(f"  ✓ Fitting successful:")
                print(f"     E0 = {params['E0']:.4f} eV")
                print(f"     V0 = {params['V0']:.4f} Å³")
                print(f"     B0 = {params['B0']:.2f} GPa")
                print(f"     B1 = {params['B1']:.3f}")
                print(f"     R² = {stats['r_squared']:.4f}")
                successful_fits += 1
            else:
                print(f"  ✗ Fitting failed")
                
        except Exception as e:
            print(f"  ✗ Error processing {structure_name}: {e}")
            import traceback
            traceback.print_exc()
    
    print(f"\nFitting summary: {successful_fits}/{total_fits} successful fits")
