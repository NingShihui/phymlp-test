import shutil
from pathlib import Path
import argparse

def classify_deform_directories(root_dir):
    """
    对形变计算目录进行分类
    
    Args:
        root_dir: 包含形变计算结果的根目录
        
    Returns:
        dict: 分类后的目录结构 {base_name: [dir_paths]}
    """
    # 匹配形变目录的模式 (以数字结尾，如_0.8, _1.0等)
    deform_pattern = re.compile(r'^(.*?)_(\d+\.\d+)$')
    
    classified = {}
    
    # 遍历根目录下的所有子目录
    for item in os.listdir(root_dir):
        item_path = os.path.join(root_dir, item)
        
        if os.path.isdir(item_path):
            match = deform_pattern.match(item)
            if match:
                base_name = match.group(1)  # 基础名称
                scale_factor = match.group(2)  # 缩放系数
                
                if base_name not in classified:
                    classified[base_name] = []
                
                classified[base_name].append({
                    'path': item_path,
                    'scale_factor': float(scale_factor)
                })
    
    return classified

def create_classified_structure(root_dir, classified_dict):
    """
    创建分类后的目录结构
    
    Args:
        root_dir: 根目录
        classified_dict: 分类后的目录字典
    """
    for base_name, dirs in classified_dict.items():
        # 创建基础目录
        base_dir = os.path.join(root_dir, base_name)
        os.makedirs(base_dir, exist_ok=True)
        
        # 按缩放系数排序
        dirs.sort(key=lambda x: x['scale_factor'])
        
        # 移动目录到新的结构
        for dir_info in dirs:
            old_path = dir_info['path']
            scale_factor = dir_info['scale_factor']
            
            # 新的目录名
            new_dir_name = f"scale_{scale_factor}"
            new_path = os.path.join(base_dir, new_dir_name)
            
            # 如果目标目录不存在，则移动
            if not os.path.exists(new_path):
                shutil.move(old_path, new_path)
                print(f"Moved: {old_path} -> {new_path}")
            else:
                print(f"Target already exists, skipping: {new_path}")

def extract_vasp_data(directory):
    """
    从VASP计算结果中提取数据
    
    Args:
        directory: 包含VASP计算结果的目录
        
    Returns:
        dict: 提取的数据 {a, V, E, P}
    """
    data = {'a': None, 'V': None, 'E': None, 'P': None}
    
    try:
        # 提取晶格常数 a
        poscar_path = os.path.join(directory, 'POSCAR')
        if os.path.exists(poscar_path):
            with open(poscar_path, 'r') as f:
                lines = f.readlines()
                if len(lines) >= 3:
                    # 提取第三行的第一个非零参数
                    third_line = lines[2].strip().split()
                    for val in third_line:
                        try:
                            num_val = float(val)
                            if num_val != 0:
                                data['a'] = round(num_val, 2)
                                break
                        except ValueError:
                            continue
        
        # 提取能量 E
        oszicar_path = os.path.join(directory, 'OSZICAR')
        if os.path.exists(oszicar_path):
            with open(oszicar_path, 'r') as f:
                lines = f.readlines()
                for line in reversed(lines):
                    if 'E0=' in line or 'F=' in line:
                        parts = line.strip().split()
                        for i, part in enumerate(parts):
                            if 'E0=' in part or 'F=' in part:
                                if i + 1 < len(parts):
                                    try:
                                        data['E'] = float(parts[i + 1])
                                        break
                                    except (ValueError, IndexError):
                                        pass
                        # 如果没找到，尝试另一种格式
                        if data['E'] is None:
                            for part in parts:
                                if 'E0=' in part:
                                    try:
                                        data['E'] = float(part.split('=')[1])
                                        break
                                    except (ValueError, IndexError):
                                        pass
        
        # 提取压力和体积
        outcar_path = os.path.join(directory, 'OUTCAR')
        if os.path.exists(outcar_path):
            with open(outcar_path, 'r', encoding='utf-8', errors='ignore') as f:
                content = f.read()
                
                # 提取外部压力
                pressure_match = re.search(r'external pressure\s*=\s*([-\d.]+)', content)
                if pressure_match:
                    try:
                        p1 = float(pressure_match.group(1))
                        data['P'] = round(p1 / 10.0, 4)  # 转换为P2
                    except ValueError:
                        pass
                
                # 提取单元格体积
                volume_match = re.search(r'volume of cell\s*:\s*([\d.]+)', content)
                if volume_match:
                    try:
                        data['V'] = float(volume_match.group(1))
                    except ValueError:
                        pass
    
    except Exception as e:
        print(f"Error extracting data from {directory}: {e}")
    
    return data

def process_all_deform_results(root_dir):
    """
    处理所有形变计算结果
    
    Args:
        root_dir: 根目录路径
    """
    print("Step 1: Classifying deform directories...")
    classified = classify_deform_directories(root_dir)
    
    if not classified:
        print("No deform directories found matching the pattern!")
        return
    
    print(f"Found {len(classified)} base structures:")
    for base_name in classified:
        print(f"  - {base_name}: {len(classified[base_name])} scale factors")
    
    print("Step 2: Creating classified structure...")
    create_classified_structure(root_dir, classified)
    
    print("Step 3: Extracting VASP data...")
    
    # 遍历新的目录结构并提取数据
    for base_name in classified.keys():
        base_dir = os.path.join(root_dir, base_name)
        
        if os.path.exists(base_dir):
            output_file = os.path.join(base_dir, 'a_V_E_P.txt')
            
            # 写入表头
            with open(output_file, 'w') as f:
                f.write("a     V     E     P\n")
            
            # 遍历所有scale目录
            scale_dirs = glob.glob(os.path.join(base_dir, "scale_*"))
            scale_dirs.sort()  # 按名称排序
            
            extracted_count = 0
            for scale_dir in scale_dirs:
                if os.path.isdir(scale_dir):
                    print(f"Processing: {scale_dir}")
                    
                    # 提取数据
                    data = extract_vasp_data(scale_dir)
                    
                    # 写入数据
                    if all(val is not None for val in data.values()):
                        with open(output_file, 'a') as f:
                            f.write(f"{data['a']}     {data['V']}     {data['E']}     {data['P']}\n")
                        print(f"  ✓ Extracted: a={data['a']}, V={data['V']}, E={data['E']}, P={data['P']}")
                        extracted_count += 1
                    else:
                        print(f"  ⚠ Warning: Incomplete data in {scale_dir}")
                        missing = [k for k, v in data.items() if v is None]
                        print(f"  Missing: {missing}")
            
            print(f"Extracted data for {extracted_count}/{len(scale_dirs)} scale factors in {base_name}")

def main():
    parser = argparse.ArgumentParser(description='Process deform calculation results')
    parser.add_argument('--root_dir', '-r', required=True, 
                       help='Root directory containing deform results')
    
    args = parser.parse_args()
    
    if not os.path.exists(args.root_dir):
        print(f"Error: Root directory '{args.root_dir}' not found!")
        return
    
    process_all_deform_results(args.root_dir)
    print("Processing completed!")



if __name__ == "__main__":
    main()
