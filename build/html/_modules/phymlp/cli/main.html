def _cmd_from_mp_structures(args):
    """从Materials Project获取结构"""
    try:
        # 添加 auto_generate_set 目录到 Python 路径
        auto_gen_dir = os.path.join(project_root, 'phymlp', 'auto_generate_set')
        if auto_gen_dir not in sys.path:
            sys.path.insert(0, auto_gen_dir)

        from from_mp_structures import main as from_mp_main
        from_mp_main(args.input_file)
    except ImportError as e:
        print(f"Error: Could not import from_mp_structures module: {e}")
        raise SystemExit(1)

        # 这里假设完整工作流在 config.py 中
        from config import main as full_workflow_main

        skip_options = {}
        if hasattr(args, 'skip_vasp_submit') and args.skip_vasp_submit:
            skip_options['skip_vasp_submit'] = True
        if hasattr(args, 'skip_extxyz') and args.skip_extxyz:
            skip_options['skip_extxyz'] = True

        full_workflow_main(args.input_file, **skip_options)
    except ImportError as e:
        print(f"Error: Could not import config module for full workflow: {e}")
        raise SystemExit(1)

def main():
    """
    PhyMLP Kit 主函数。

    处理所有命令行参数并执行相应子命令。
    """
    parser = ArgumentParser(
        description="PhyMLP: A Python Package for PhyMLP Training and Validation",
        epilog="For more information about PhyMLP commands, use phymlp-kit <command> -h"
    )

    subparsers = parser.add_subparsers(
        dest="command",
        title="Sub-commands",
        description="Choose a command to execute",
        required=True
    )

    # 1. train
    parser_train = subparsers.add_parser(
        "train",
        help="Train and validate PhyMLP, e.g. phymlp-kit train --train_input Inputfile"
    )
    parser_train.add_argument("--train_input", help="Input file for training", required=True)

    # 2. generate_kpath
    parser_gk = subparsers.add_parser(
        "generate_kpath",
        help="Generate KPATH for unit-POSCAR, e.g. phymlp-kit generate_kpath --vaspkit_load xx/vaspkit"
    )
    parser_gk.add_argument("--vaspkit_load", help="Path to vaspkit executable", required=True)

    # 3. init_phymlp
    parser_init = subparsers.add_parser(
        "init_phymlp",
        help="Get initial init.PhyMLP, e.g. phymlp-kit init_phymlp --phymlp_load xx/phymlp"
    )
    parser_init.add_argument("--phymlp_load", help="Path to phymlp-kit install", required=True)
    parser_init.add_argument(
        "--complexity",
        choices=["06.PhyMLP", "08.PhyMLP", "10.PhyMLP", "12.PhyMLP", "14.PhyMLP", "16.PhyMLP"],
        default="10.PhyMLP"
    )

    # 4. outcar2extxyz
    parser_oc = subparsers.add_parser(
        "outcar2extxyz",
        help="Convert OUTCAR to extxyz, e.g. phymlp-kit outcar2extxyz --outcar2extxyz_input ./OUTCAR"
    )
    parser_oc.add_argument("--outcar2extxyz_input", default="./OUTCAR", help="OUTCAR file path")
    parser_oc.add_argument("--outcar2extxyz_output", default="./train.extxyz", help="Output extxyz file path")
    parser_oc.add_argument("--step", type=int, default=1, help="Step for extracting structures")

    # 5. cfg2extxyz
    parser_cfg = subparsers.add_parser(
        "cfg2extxyz",
        help="Convert cfg to extxyz, e.g. phymlp-kit cfg2extxyz --cfg2extxyz_input ./train.cfg --elements Ni Cr ..."
    )
    parser_cfg.add_argument("--cfg2extxyz_input", default="./train.cfg", help=".cfg file path")
    parser_cfg.add_argument("--cfg2extxyz_output", default="./train.extxyz", help="Output extxyz file path")
    parser_cfg.add_argument("--elements", nargs='+', help="The mapping from atom types to elements", required=True)

    # 6. split_set
    parser_split = subparsers.add_parser(
        "split_set",
        help="Split set to set_train and set_validation"
    )
    parser_split.add_argument("--input_file", type=str, required=True, help='Input extxyz file')
    parser_split.add_argument("--set_train", type=str, default='train.extxyz', help='Name of training set')
    parser_split.add_argument("--set_val", type=str, default='validation.extxyz', help='Name of validation set')
    parser_split.add_argument("--val_ratio", type=float, default=0.1, help='Ratio of validation set')

    # 7. 新增：从Materials Project获取结构
    parser_mp = subparsers.add_parser(
        "from_mp_structures",
        help="Get structures from Materials Project, e.g. phymlp-kit from_mp_structures --input_file input.yaml"
    )
    parser_mp.add_argument("--input_file", required=True, help="YAML configuration file")

    # 8. 新增：生成形变结构
    parser_deform = subparsers.add_parser(
        "deform_structures",
        help="Generate deformed structures, e.g. phymlp-kit deform_structures --input_file input.yaml"
    )
    parser_deform.add_argument("--input_file", required=True, help="YAML configuration file")

    # 9. 新增：生成微扰结构
    parser_perturb = subparsers.add_parser(
        "perturb_structures",
        help="Generate perturbed structures, e.g. phymlp-kit perturb_structures --input_file input.yaml"
    )
    parser_perturb.add_argument("--input_file", required=True, help="YAML configuration file")

    # 10. 新增：设置VASP计算
    parser_setup = subparsers.add_parser(
        "setup_vasp",
        help="Setup VASP calculations, e.g. phymlp-kit setup_vasp --input_file input.yaml"
    )
    parser_setup.add_argument("--input_file", required=True, help="YAML configuration file")

    # 11. 新增：提交VASP计算
    parser_submit = subparsers.add_parser(
        "submit_vasp",
        help="Submit VASP calculations, e.g. phymlp-kit submit_vasp --input_file input.yaml"
    )
    parser_submit.add_argument("--input_file", required=True, help="YAML configuration file")

    # 12. 新增：测试POTCAR生成
    parser_potcar = subparsers.add_parser(
        "test_potcar",
        help="Test POTCAR generation, e.g. phymlp-kit test_potcar --poscar POSCAR --input_file input.yaml"
    )
    parser_potcar.add_argument("--poscar", required=True, help="POSCAR file path")
    parser_potcar.add_argument("--input_file", required=True, help="YAML configuration file")

    # 13. 新增：处理形变结果
    parser_process = subparsers.add_parser(
        "process_deform",
        help="Process deformation results, e.g. phymlp-kit process_deform --input_file input.yaml"
    )
    parser_process.add_argument("--input_file", required=True, help="YAML configuration file")

    # 14. 新增：Birch-Murnaghan拟合
    parser_bm = subparsers.add_parser(
        "birch_murnaghan",
        help="Perform Birch-Murnaghan fitting, e.g. phymlp-kit birch_murnaghan --input_file input.yaml"
    )
    parser_bm.add_argument("--input_file", required=True, help="YAML configuration file")

    # 15. 新增：生成Extxyz数据集
    parser_extxyz = subparsers.add_parser(
        "generate_extxyz",
        help="Generate extxyz dataset, e.g. phymlp-kit generate_extxyz --input_file input.yaml"
    )
    parser_extxyz.add_argument("--input_file", required=True, help="YAML configuration file")

    # 16. 新增：执行完整工作流
    parser_workflow = subparsers.add_parser(
        "full_workflow",
        help="Execute full workflow, e.g. phymlp-kit full_workflow --input_file input.yaml"
    )
    parser_workflow.add_argument("--input_file", required=True, help="YAML configuration file")
    parser_workflow.add_argument("--skip_vasp_submit", action="store_true", help="Skip VASP submission step")
    parser_workflow.add_argument("--skip_extxyz", action="store_true", help="Skip extxyz generation step")

    # 17. 新增：实验P-V数据拟合
    parser_exp_pv = subparsers.add_parser(
        "exp_pv_structures",
        help="Fit experimental P-V data and generate structures, e.g. phymlp-kit exp_pv_structures --input_file input.yaml"
    )
    parser_exp_pv.add_argument("--input_file", required=True, help="YAML configuration file")

    # 18. 新增：dump数据拟合
    parser_dump_to_poscar = subparsers.add_parser(
        "dump_to_poscar",
        help="dump data to poscar, e.g. phymlp-kit dump_to_poscar --input_file input.yaml"
    )
    parser_dump_to_poscar.add_argument("--input_file", required=True, help="YAML configuration file")

    # ---------------- 解析并执行 ----------------
    args = parser.parse_args()

    if args.command == "generate_kpath":
        generate_kpath(os.path.abspath(args.vaspkit_load))

    elif args.command == "init_phymlp":
        subprocess.run(f"cp {os.path.abspath(args.phymlp_load)}/PhyMLP_init/{args.complexity} ./",
                       shell=True).check_returncode()

    elif args.command == "outcar2extxyz":
        out = Path(args.outcar2extxyz_output)
        out.parent.mkdir(parents=True, exist_ok=True)
        extract_structures_from_outcar(args.outcar2extxyz_input, str(out), args.step)

    elif args.command == "cfg2extxyz":
        out = Path(args.cfg2extxyz_output)
        out.parent.mkdir(parents=True, exist_ok=True)
        configs = parse_cfg_file(args.cfg2extxyz_input)
        type_map = {i: el for i, el in enumerate(args.elements)}
        write_extxyz(configs, str(out), type_map=type_map, float_fmt=".6f")

    elif args.command == "split_set":
        split_extxyz_structures(args.input_file, args.set_train, args.set_val, args.val_ratio)

    elif args.command == "train":
        run = RUN()
        info = run.Read(args.train_input)
        run.Key_and_parament(info)
        bool_train = run.Bool_commands(run.train)
        train_dir, train_cmd = run.Train_commands(bool_train)
        run.Train(bool_train, train_dir, train_cmd)
        bool_calc = run.Bool_commands(run.calculate)
        run.Calculate_dir(bool_calc)
        run.Rewrite_lammps_in(bool_calc)
        run.Execute(bool_calc)
        print("----------------------所有指定的计算任务均已完成！-----------------------")

    # ===== 新增：自动训练集生成相关命令 =====
    elif args.command == "from_mp_structures":
        _cmd_from_mp_structures(args)

    elif args.command == "deform_structures":
        _cmd_deform_structures(args)

    elif args.command == "perturb_structures":
        _cmd_perturb_structures(args)

    elif args.command == "setup_vasp":
        _cmd_setup_vasp(args)

    elif args.command == "submit_vasp":
        _cmd_submit_vasp(args)

    elif args.command == "test_potcar":
        _cmd_test_potcar(args)

    elif args.command == "process_deform":
        _cmd_process_deform(args)

    elif args.command == "birch_murnaghan":
        _cmd_birch_murnaghan(args)

    elif args.command == "exp_pv_structures":
        _cmd_exp_pv_structures(args)

    elif args.command == "generate_extxyz":
        _cmd_generate_extxyz(args)

    elif args.command == "dump_to_poscar":
        _cmd_dump_to_poscar(args)

    elif args.command == "full_workflow":
        _cmd_full_workflow(args)

    else:
        parser.print_help()
        raise SystemExit(1)




if __name__ == "__main__":
    main()
