    def Read(self, Inputfile):
        """读取Inputfile配置文件"""
        input_informations = []
        try:
            with open(Inputfile, 'r', encoding='utf-8') as readfile:
                read_informations = readfile.readlines()
                for read_information in read_informations:
                    new_line = "".join(read_information.split())
                    if new_line.startswith('#') or len(new_line) == 0:
                        continue
                    new_line = new_line.split("#")[0]
                    input_informations.append(new_line)
                return input_informations
        except Exception as e:
            print(f"文件读取错误: {e}")
            sys.exit(1)



    def _parse_value(self, value_str):
        """安全地解析配置值"""
        value_str = value_str.strip()

        # 处理字符串类型的值（带引号的）
        if (value_str.startswith('"') and value_str.endswith('"')) or \
                (value_str.startswith("'") and value_str.endswith("'")):
            return value_str[1:-1]

        # 尝试解析为Python字面量
        try:
            parsed_value = ast.literal_eval(value_str)
            return parsed_value
        except (ValueError, SyntaxError):
            # 如果解析失败，返回原始字符串
            return value_str

    def Key_and_parament(self, input_informations):
        """解析Inputfile中的参数并设置类属性"""
        print("############识别到的设置的Inputfile中的指令#################")

        # 创建属性名到配置键的映射（处理大小写和命名差异）
        key_mapping = {
            'phymlp_load': 'PhyMLP_load',
            'lmp_mpi_load': 'lmp_mpi_load',
            'train': 'train',
            'extxyz_load': 'extxyz_load',
            'init_phymlp': 'init_PhyMLP',  # 处理大小写差异
            'iteration_limit': 'iteration_limit',
            'al_mode': 'al_mode',
            'save_to': 'save_to',
            'pot_phymlp_load': 'pot_PhyMLP_load',
            'calculate': 'calculate',
            'properties': 'properties',
            'crystal_structure': 'crystal_structure',
            'lattice': 'lattice',
            'atomic_mass': 'atomic_mass',
            'vaspkit_load': 'vaspkit_load'
        }

        temp_properties = []

        for input_information in input_informations:
            # 分割键值对
            if '=' not in input_information:
                print(f"警告: 跳过无效的配置行: {input_information}")
                continue

            parts = input_information.split('=', 1)
            raw_key = parts[0].strip()
            input_key = raw_key.lower()  # 转换为小写进行比较
            value_str = parts[1].strip()

            # 检查键是否在允许的列表中
            if input_key in key_mapping:
                # 获取正确的属性名
                attribute_name = key_mapping[input_key]

                if attribute_name == "properties":
                    # 处理properties参数
                    property_value = self._parse_value(value_str)

                    if isinstance(property_value, list):
                        for prop in property_value:
                            if prop in self.all_properties:
                                temp_properties.append(prop)
                            else:
                                print(f"不支持的计算属性: {prop}")
                                sys.exit(1)
                        self.properties = temp_properties
                    elif isinstance(property_value, str):
                        props = [p.strip() for p in property_value.split(',')]
                        for prop in props:
                            if prop in self.all_properties:
                                temp_properties.append(prop)
                            else:
                                print(f"不支持的计算属性: {prop}")
                                sys.exit(1)
                        self.properties = temp_properties
                    else:
                        print(f"properties参数格式错误: {property_value}")
                        sys.exit(1)
                else:
                    # 解析值并设置属性
                    parsed_value = self._parse_value(value_str)
                    setattr(self, attribute_name, parsed_value)

                print(f"{raw_key}={value_str}")
            else:
                print(f"不支持的指令: {raw_key}")
                sys.exit(1)

        print("#############未设置的指令将保持默认#########################")

        # 打印更新后的参数用于调试
        self._print_updated_parameters()

    def slurm_sub(self, job_name, command=""):
        """提交SLURM作业

        Args:
            job_name (str): 作业名称
            command (str, optional): 训练命令，仅用于训练作业

        Raises:
            subprocess.CalledProcessError: 当作业提交失败时
            RuntimeError: 当作业监控过程中出现错误时
        """
        if str(job_name).lower() == "train":
            job_script = f"""#!/bin/bash
#SBATCH -J calculate_{job_name}
#SBATCH -N 1
#SBATCH -p xahcnormal
#SBATCH --ntasks-per-node=32
mpirun -n 32 {command} > train-output.log
            """
        else:
            job_script = f"""#!/bin/bash
#SBATCH -J calculate_{job_name}
#SBATCH -N 1
#SBATCH -p xahcnormal
#SBATCH --ntasks-per-node=32
mpirun -n 32 {self.lmp_mpi_load} -i {job_name}.in > output.log
            """

        current_path = os.getcwd()
        job_dir = os.path.join(current_path, str(job_name))
        os.makedirs(job_dir, exist_ok=True)

        script_path = os.path.join(job_dir, "test_job_script.sh")
        with open(script_path, "w") as f:
            f.write(job_script)

        process = subprocess.Popen(
            f"cd {job_dir} && sbatch test_job_script.sh",
            stdout=subprocess.PIPE,
            stderr=subprocess.PIPE,
            shell=True
        )
        process.wait()
        stdout, stderr = process.communicate()

        if process.returncode != 0:
            print(f"作业提交失败: {stderr.decode()}")
            return

        job_id = stdout.decode().strip().split()[-1]
        print(f"计算 {job_name} 任务已经提交，作业号: {job_id}")
        print("运行中......")

        self._wait_for_job_completion(job_id, job_name, current_path, job_dir)



    def _wait_for_job_completion(self, job_id, job_name, current_path, job_dir):
        """等待作业完成并执行后处理

        Args:
            job_id (str): SLURM作业ID
            job_name (str): 作业名称
            current_path (str): 当前工作目录
            job_dir (str): 作业目录
        """
        waiting_event = threading.Event()

        def waiting(event):
            chars = ['\\', '|', '/', '-']
            i = 0
            while not event.is_set():
                print(f'\r{chars[i]}', end='', flush=True)
                i = (i + 1) % len(chars)
                time.sleep(0.1)

        try:
            waiting_thread = threading.Thread(target=waiting, args=(waiting_event,))
            waiting_thread.start()

            while True:
                check_process = subprocess.Popen(
                    ["squeue", "-j", job_id],
                    stdout=subprocess.PIPE,
                    stderr=subprocess.PIPE
                )
                check_stdout, check_stderr = check_process.communicate()

                if job_id not in check_stdout.decode():
                    self._post_process(job_name, current_path)
                    break
                elif "R" in check_stdout.decode() or "PD" in check_stdout.decode():
                    time.sleep(5)

            waiting_event.set()
            waiting_thread.join()

        except Exception as e:
            print(f"作业等待错误: {e}")

        os.remove(os.path.join(job_dir, "test_job_script.sh"))

    def _post_process(self, job_name, current_path):
        """作业完成后处理

        Args:
            job_name (str): 作业名称
            current_path (str): 当前工作目录
        """
        job_name_lower = job_name.lower()

        if job_name_lower in ["e-v", "ev"]:
            plot_command = f"cd {current_path}/{job_name} && python plot.py && rm plot.py"
            subprocess.Popen(plot_command, shell=True).wait()

        elif job_name_lower in ["p-v", "pv"]:
            plot_command = f"cd {current_path}/{job_name} && python plot.py && rm plot.py"
            subprocess.Popen(plot_command, shell=True).wait()

        elif job_name_lower == "phonon":
            self._process_phonon(current_path, job_name)

        print(f"计算已经完成，请到 {job_name} 目录下查看结果")

    def _process_phonon(self, current_path, job_name):
        """处理声子计算的后处理

        Args:
            current_path (str): 当前工作目录
            job_name (str): 作业名称
        """
        commands = [
            "phonolammps PHONON.in -c POSCAR --dim 2 2 2",
            "python get_KPATH.py >> PHONON.log",
            "rm ./get_KPATH.py",
            "sed -i 's/DIM =/DIM = 2 2 2/' ./KPATH.phonopy",
            "phonopy -c POSCAR -p -s KPATH.phonopy"
        ]
        full_command = " && ".join(commands)
        plot_command = f"cd {current_path}/{job_name} && {full_command}"
        subprocess.Popen(plot_command, shell=True).wait()

    def Train_commands(self, drive):
        """生成训练命令和目录

        Args:
            drive (bool): 是否执行训练

        Returns:
            tuple: (训练目录路径, 训练命令) 或 ("No_path", "Skip_step")

            f"mkdir -p train && cp -f {self.init_PhyMLP_load}{self.init_PhyMLP} {self.extxyz_load} ./train",

            sys.exit(1)

        print("在当前目录下创建了用于训练PhyMLP势的train子目录")

        train_dir = os.path.join(self.current_path, "train")
        train_command = (
            f"{self.PhyMLP_load}/bin/MLP train {self.init_PhyMLP} {self.extxyz_load} "
            f"--save_to={self.save_to} --iteration_limit={self.iteration_limit} "
            f"--al_mode={self.al_mode}"
        )

        return train_dir, train_command

    def Rewrite_lammps_in(self, drive):
        """根据配置重写LAMMPS输入文件

        Args:
            drive (bool): 是否执行计算任务
        """
        if not drive:
            return

        for calculate_property in self.properties:
            calculate_property = str(calculate_property).upper()
            self._modify_property_in_file(calculate_property)



    def _modify_property_in_file(self, property):
        """修改特定属性的输入文件

        Args:
            property (str): 计算属性名称

        Raises:
            SystemExit: 当文件修改失败时
        """
        property_commands = {
            "E-V": self._modify_ev_in,
            "ELASTIC": self._modify_elastic_in,
            "P-V": self._modify_pv_in,
            "PHONON": self._modify_phonon_in
        }

        if property in property_commands:
            property_commands[property]()
        else:
            print(f"不支持的属性: {property}")

    def _modify_ev_in(self):
        """修改E-V计算输入文件"""
        lattice_min = '%.2f' % (float(self.lattice) - float(self.lattice) * 0.3)
        interval = '%.2f' % (2 * float(self.lattice) * 0.3 / 100)

        commands = [
            f"sed -i 's/define_mass/mass 1 {self.atomic_mass}/' ./E-V/E-V.in",
            f"sed -i 's/crystal_structure/{self.crystal_structure}/' ./E-V/E-V.in",
            f"sed -i 's/lattice_constant/{lattice_min}+{interval}*$i/' ./E-V/E-V.in"
        ]
        self._execute_commands(commands, "E-V")

    def _modify_elastic_in(self):
        """修改弹性常数计算输入文件"""

    def _execute_commands(self, commands, property_name):
        """执行命令列表

        Args:
            commands (list): 命令字符串列表
            property_name (str): 属性名称，用于错误信息

        Raises:
            SystemExit: 当命令执行失败时
        """
        full_command = " && ".join(commands)
        process = subprocess.Popen(full_command, shell=True)
        process.wait()

        if process.returncode != 0:
            print(f"请检查{property_name}计算参数是否设置正确")
            sys.exit(1)
